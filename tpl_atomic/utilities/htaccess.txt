##################################################################
## 1. SECURITY HEADERS
##################################################################

<IfModule mod_headers.c>
  Header always set X-Content-Type-Options "nosniff"
  Header always set X-Frame-Options "SAMEORIGIN"
  Header always set Referrer-Policy "strict-origin-when-cross-origin"
  Header always set Permissions-Policy "geolocation=(), camera=(), microphone=()"
  # Uncomment HSTS only after confirming the full site (and subdomains you care
  # about) are served exclusively over HTTPS:
  # Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains"
</IfModule>

## Disable inline JavaScript when SVG files are opened directly
<FilesMatch "\.svg$">
  <IfModule mod_headers.c>
    Header always set Content-Security-Policy "script-src 'none'"
  </IfModule>
</FilesMatch>

## Disable directory listings (belt-and-suspenders with Joomla's Options -Indexes)
<IfModule mod_autoindex.c>
  IndexIgnore *
</IfModule>


##################################################################
## 2. BLOCK COMMON EXPLOIT / RECON PATHS
##################################################################

# Block WordPress-style scans (this is a Joomla site)
RedirectMatch 404 ^/wp-
RedirectMatch 404 ^/wordpress
RedirectMatch 404 ^/xmlrpc\.php$

# Block source-control / build-tool file exposure
RedirectMatch 404 /\.git(/|$)
RedirectMatch 404 /\.svn(/|$)
RedirectMatch 404 /\.hg(/|$)
RedirectMatch 404 /\.bzr(/|$)
RedirectMatch 404 /\.env$
RedirectMatch 404 /\.DS_Store$
RedirectMatch 404 /composer\.(json|lock)$
RedirectMatch 404 /package(-lock)?\.json$
RedirectMatch 404 /yarn\.lock$
RedirectMatch 404 /Gemfile(\.lock)?$
RedirectMatch 404 /phpunit(\.xml(\.dist)?)?$

# Block PHP files under /.well-known/ (common backdoor-planting spot, e.g. nastar.php)
RedirectMatch 403 ^/\.well-known/.*\.php$

# Block probes for temp, backup, and staging directories
RedirectMatch 403 ^/(backup|backups|bak|old|db|database|dump|dumps|sql|mysql|temp|tmp|test|tests|dev|staging|stage|new|beta)/?$

# Block direct access to Joomla sensitive files
<FilesMatch "^(configuration\.php|\.htaccess|\.htpasswd|web\.config|htaccess\.txt)$">
  Require all denied
</FilesMatch>

# Block targeted Joomla Astroid installTemplate exploit
<IfModule mod_rewrite.c>
  RewriteEngine On
  RewriteCond %{QUERY_STRING} astroid=installTemplate [NC]
  RewriteRule .* - [F,L]

  # Block common SQLi / RFI / LFI patterns in query strings
  RewriteCond %{QUERY_STRING} (\.\./){2,}            [NC,OR]
  RewriteCond %{QUERY_STRING} (union[\s+]+select)    [NC,OR]
  RewriteCond %{QUERY_STRING} (concat[\s+]*\()       [NC,OR]
  RewriteCond %{QUERY_STRING} (benchmark[\s+]*\()    [NC,OR]
  RewriteCond %{QUERY_STRING} (sleep[\s+]*\([\s+]*[0-9]) [NC,OR]
  RewriteCond %{QUERY_STRING} (passthru|shell_exec|system|phpinfo)[\s+]*\( [NC,OR]
  RewriteCond %{QUERY_STRING} (\bfile_get_contents\b|\bfopen\b)\( [NC,OR]
  RewriteCond %{QUERY_STRING} (=|%3D)(php|https?|ftp|data|file)(:|%3A)// [NC]
  RewriteRule .* - [F,L]

  # Block requests with no User-Agent AND no Referer hitting dynamic endpoints
  # (legitimate browsers always send a UA)
  RewriteCond %{HTTP_USER_AGENT} ^-?$
  RewriteCond %{REQUEST_URI}     \.(php|cgi|pl)$ [NC]
  RewriteRule .* - [F,L]
</IfModule>

##################################################################
## 5. BROWSER CACHING (SEO & Page Speed)
##################################################################

# Three months for image files
<filesMatch ".(jpg|jpeg|png|gif|ico|webp|svg)$">
  Header set Cache-Control "max-age=7884000, public"
</filesMatch>

# One month for CSS and JS
<filesMatch ".(css|js)$">
  Header set Cache-Control "max-age=2628000, public"
</filesMatch>

# One year for fonts
<filesMatch ".(woff|woff2|ttf|otf|eot)$">
  Header set Cache-Control "max-age=31536000, public"
</filesMatch>

# Seven days for other files
<filesMatch ".(pdf|flv|swf)$">
  Header set Cache-Control "max-age=604800, public"
</filesMatch>
